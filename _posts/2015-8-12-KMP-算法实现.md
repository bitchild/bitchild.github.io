### 求解 NEXT 数组

    //len_mod is the length of pattern string
	//mod is the pattern string

	int		NEXT[len_mod+1]; 
	int		p = 1, q = 0;
	
	NEXT[0] = -1;
	NEXT[1] = 0;
	
	while (p < len_mod)
	{
		if (mod[p] == mod[q])
		NEXT[++p] = ++q;
		else
		{
			if ((q = NEXT[q]) == -1)
			NEXT[++p] = ++q;
		}
	}
